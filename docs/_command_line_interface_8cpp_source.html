76 CommandLineInterface& CommandLineInterface::process()
77 {
78  foreach( QString a, m_inputArguments )
79  {
80  InputArgument arg( a, m_acceptedArgumentPrefixs );
81  if ( arg.isValid() )
82  {
83  QString key = arg.name().toLower();
84  if ( m_arguments.contains( key ) )

86  m_arguments[key]->callback()(arg.value());

88  }
89  }

188 
189  QStringList filterArgumentAliases;
190  foreach( Argument* arg, m_arguments.values() )
191  {
192  if ( filterArgumentAliases.contains( arg->name() ) ) continue;
193  if ( arg->hasShortName() )

195  message += QString( " -%1, --%2").arg(arg->shortName()).arg(arg->name());
196  }
197  else
198  {
199  message += QString( " -%1").arg(arg->name());
200  }
201

204  }
