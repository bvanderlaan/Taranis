76 CommandLineInterface& CommandLineInterface::process()
77 {
78  int numOfArguments = m_inputArguments.count();
79  for( int i = 0; i < numOfArguments; ++i )
80  {
81  InputArgument* arg = new InputArgument( m_inputArguments.at(i), m_acceptedArgumentPrefixs );
82  if ( ( numOfArguments > 1 ) && (i < numOfArguments - 1 ) && ( arg->isValid()) )
83  {
84  bool isNextArgumentValid = InputArgument( m_inputArguments.at(i+1), m_acceptedArgumentPrefixs ).isValid();
85  if ( !isNextArgumentValid )
86  {
87  QString multiPartInput = QString("%1%2%3")
88  .arg(m_inputArguments.at(i))
89  .arg( arg->nameValueSeperator().isEmpty() ? ":" : "" )
90  .arg(m_inputArguments.at(i+1));
91 
92  delete arg;
93  arg = new InputArgument( multiPartInput , m_acceptedArgumentPrefixs );
94  ++i;
95  }
96  }
97 
98  if ( arg->isValid() )
99  {
100  QString key = arg->name().toLower();
101  if ( m_arguments.contains( key ) )
102  {
103  m_arguments[key]->callback()(arg->value());
104  }
105  }
106  delete arg;
107  }
108 
109  return *this;
110 }
