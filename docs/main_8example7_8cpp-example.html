This example shows how to define a command line interface which contains an argument which accepts values.

If you run this example when the user provides the server argument such as –server 1.2.3.4, –server=1.2.3.4, –server:1.2.3.4, –server= 1.2.3.4, or –server: 1.2.3.4 you will see the output Server IP: 1.2.3.4.

#ifdef EXAMPLE7
#include <QCoreApplication>
#include <QDebug>
#include "CommandLineInterface.hpp"
using namespace Taranis;
int main(int argc, char *argv[])
{
QCoreApplication a(argc, argv);
CommandLineInterface arguments = CommandLineInterface::build()
.WithName("My Cool App")
.WithVersion("1.2.3.4-abc")
.WithValue("server", "Lets you define the IP of the server.");
qDebug() << "Server IP: " << ( arguments["server"].isValid() ? arguments["server"].toString() : "Unknown" );
return a.exec();
}
#endif
